## Check whether OS in Windows or Linux
```
from sys import platform # check platform OS Windows or Linux to adapt / - \
# check platform OS Windows or Linux, adapt paths accordingly: 
if platform == 'win32':
	ROOT = 'C:'
	EXE = 'fgdfsfg.exe'
	SLASH = '\\'
else: # Linux
	ROOT = os.getenv("HOME") 
	EXE = 'dfsdsgg'
	SLASH = '/'
```
